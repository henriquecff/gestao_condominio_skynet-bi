SELECT cast(concat(fatto.ID_PROPRIETARIO, fatto.ID_MOVIMENTACAO, fatto.ID_LOTE, fatto.ID_CONDOMINIO) as bigint) as key_dms,
		fatto.dt_movimentacao, fatto.dia_semana, fatto.quinzena, fatto.trimestre, fatto.semestral, fatto.ano, fatto.ano_mes
FROM

(
	select prop.ID_PROPRIETARIO, mvt.ID_MOVIMENTACAO, lt.ID_LOTE, cdm.ID_CONDOMINIO,
       mvt.dt_movimentacao,
	   datepart(weekday, mvt.dt_movimentacao) as dia_semana,
	   mvt.tp_movimentacao, 
	   case when day(mvt.dt_movimentacao) <= 15 then 1 else 2 end as quinzena,
	   concat(year(mvt.dt_movimentacao), month(mvt.dt_movimentacao)) as ano_mes,
	   case 
		when month(mvt.dt_movimentacao) <= 3 then 
			1
		when month(mvt.dt_movimentacao) > 3 and month(mvt.dt_movimentacao) <= 6 then 
			2
		when month(mvt.dt_movimentacao) > 6 and month(mvt.dt_movimentacao) <= 9 then 
			3
		else
			4
	   end as trimestre,
	   case 
		when month(mvt.dt_movimentacao) <= 6 then 
			1
		else
			2
	   end as semestral,
	   month(mvt.dt_movimentacao) as mes,
	   year(mvt.dt_movimentacao) ano

		from dbo.T_GCD_PROPRIETARIO prop
		inner join dbo.T_GCD_LOTE_PROP lt_prd
		on lt_prd.id_proprietario = prop.id_proprietario
		inner join dbo.T_GCD_MOVIMENTACAO_PORTARIA mvt
		on mvt.id_lote_prop = lt_prd.id_lote_prop
		inner join dbo.T_GCD_LOTE lt
		on lt.id_lote = lt_prd.id_lote
		inner join dbo.T_GCD_CONDOMINIO cdm
		on cdm.id_condominio = lt.id_condominio
		
) as fatto;

SELECT 
		cast(concat(fatto.ID_PROPRIETARIO, fatto.ID_MOVIMENTACAO, fatto.ID_LOTE, fatto.ID_CONDOMINIO) as bigint) as ID_FATO,
		cast(concat(fatto.ID_PROPRIETARIO, fatto.ID_MOVIMENTACAO, fatto.ID_LOTE, fatto.ID_CONDOMINIO) as bigint) as ID_TEMPO,
		cast(concat(fatto.ID_PROPRIETARIO, fatto.NR_CPF) as bigint) as ID_DM_PROPRIETARIO,
		cast(concat(fatto.ID_PROPRIETARIO, fatto.ID_MOVIMENTACAO, fatto.ID_LOTE, fatto.ID_CONDOMINIO) as bigint) as ID_DM_MOVIMENTACAO,
		cast(concat(fatto.ID_CONDOMINIO, fatto.NR_CNPJ) as bigint) as ID_DM_CONDOMINIO,
		cast(concat(fatto.ID_PROPRIETARIO, fatto.ID_LOTE, fatto.ID_CONDOMINIO) as bigint) as ID_DM_LOTE,
	   count(fatto.dt_movimentacao) as total_mvt_dia, count(fatto.quinzena) as total_mvt_quinzena, 
	   count(fatto.trimestre) as total_mvt_trimestre, count(fatto.semestral) as total_mvt_semestre, count(fatto.ano) total_mvt_ano
		
FROM

(
	select prop.ID_PROPRIETARIO, mvt.ID_MOVIMENTACAO, lt.ID_LOTE, cdm.ID_CONDOMINIO, prop.NR_CPF, cdm.NR_CNPJ,
       mvt.dt_movimentacao,
	   datepart(weekday, mvt.dt_movimentacao) as dia_semana,
	   mvt.tp_movimentacao, 
	   case when day(mvt.dt_movimentacao) <= 15 then 1 else 2 end as quinzena,
	   concat(year(mvt.dt_movimentacao), month(mvt.dt_movimentacao)) as ano_mes,
	   case 
		when month(mvt.dt_movimentacao) <= 3 then 
			1
		when month(mvt.dt_movimentacao) > 3 and month(mvt.dt_movimentacao) <= 6 then 
			2
		when month(mvt.dt_movimentacao) > 6 and month(mvt.dt_movimentacao) <= 9 then 
			3
		else
			4
	   end as trimestre,
	   case 
		when month(mvt.dt_movimentacao) <= 6 then 
			1
		else
			2
	   end as semestral,
	   month(mvt.dt_movimentacao) as mes,
	   year(mvt.dt_movimentacao) ano

		from dbo.T_GCD_PROPRIETARIO prop
		inner join dbo.T_GCD_LOTE_PROP lt_prd
		on lt_prd.id_proprietario = prop.id_proprietario
		inner join dbo.T_GCD_MOVIMENTACAO_PORTARIA mvt
		on mvt.id_lote_prop = lt_prd.id_lote_prop
		inner join dbo.T_GCD_LOTE lt
		on lt.id_lote = lt_prd.id_lote
		inner join dbo.T_GCD_CONDOMINIO cdm
		on cdm.id_condominio = lt.id_condominio
		
) as fatto
group by fatto.ID_PROPRIETARIO, fatto.ID_MOVIMENTACAO, fatto.ID_LOTE, fatto.ID_CONDOMINIO, fatto.ano_mes, fatto.tp_movimentacao, fatto.dt_movimentacao, fatto.quinzena, fatto.trimestre, fatto.semestral, fatto.ano, fatto.NR_CPF, fatto.NR_CNPJ
